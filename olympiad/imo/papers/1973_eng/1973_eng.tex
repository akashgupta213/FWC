\section* {1973/1.}

\begin{enumerate}
    

\item  Point $O$ lies on line $g$.  
Vectors $\overrightarrow{OP_1}, \overrightarrow{OP_2}, \ldots, \overrightarrow{OP_n}$ are unit vectors such that the points  
\item  $P_1, P_2, .... , P_n$ all lie in a plane containing $g$ and on one side of $g$.  

Prove that if $n$ is odd, then

\begin{align}
\left | \overrightarrow{OP_1} + \overrightarrow{OP_2} + ... + \overrightarrow{OP_n} \right| \ge 1
\end{align}

Here $\left | \overrightarrow{OM} \right |$ denotes the length of the vector $\overrightarrow{OM}$

\end{enumerate}

\section* {1973/2.}
Determine whether or not there exists a finite set $M$ of points in space not
lying in the same plane such that, for any two points $A$ and $B$ of $M$, one can
select two other points $C$ and $D$ of $M$ so that lines $AB$ and $CD$ are parallel
and not coincident.

\section* {1973/3.}
Let $a$ and $b$ be real numbers for which the equation
    
\begin{align}
    x_4 + ax_3 + bx_2 + ax + 1 = 0
\end{align}has at least one real solution. For all such pairs $(a, b)$, find the minimum
value of $a^2 + b^2$
.

\section* {1973/4.}
A soldier needs to check on the presence of mines in a region having the
shape of an equilateral triangle. The radius of action of his detector is equal
to half the altitude of the triangle. The soldier leaves from one vertex of the
triangle. What path shouid he follow in order to travel the least possible
distance and still accomplish his mission?


\section* {1972/5.}
$G$ is a set of non-constant functions of the real variable $x$ of the form

\begin{align}
    f(x) = ax + b, a and b are real numbers,
\end{align}and G has the following properties:

\begin{enumerate}
    \item If $f$ and $g$ are in $G$, then $g$ $o$ $f$ is in $G$; here ($g$ $o$ $f$)$(x)$ = $g[f(x)]$.
\item If $f$ is in $G$, then its inverse $f^{-1}$ is in $G$; here the inverse of $f(x)=ax+b$ is $f^{-1}(x)={x-b}/{a}$.


\item For every $f$ in $G$, there exists a real number $x_f$ such that $f(x_f)$ = $x_f$ .
Prove that there exists a real number k such that $f(k)$ = $k$ for all $f$ in $G$.

\end{enumerate}Prove that there exists a real number $k$ such that $f(k) = k$ for all f in $G$.


\section*{1972/6.}

Given four distinct parallel planes, prove that there exists a regular tetrahedron with a vertex on each plane.


Let $a_1,a_2,...,a_n$ be positive numbers, and let $q$ be a real number such that $0<q<1$.  
Find numbers $b_1,b_2,...,b_n$ such that

\begin{enumerate}
\item $a_k<b_k$ for $k=1,2,...,n,$

\item $q<\frac{b_k}{b_{k+1}}< \frac{1}{q}$ for $k=1,2,...,n-1,$

\item $b_1+b_2+...+b_n <  \frac{1+q}{1-q}\,(a_1+a_2+...+a_n)$
\end{enumerate}
